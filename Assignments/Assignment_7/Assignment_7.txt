library(tidyverse)library(janitor)library(plotly)library(formattable)library(htmlwidgets)df <- read_csv("./Utah_Religions_by_County.csv")df <- df %>%    select(-Religious) %>%    pivot_longer(-c(County, Pop_2010),                 names_to = "religion",                 values_to = "religion_percent") %>%    clean_names()# “Does population of a county correlate with the proportion of any specific religious group in that county?”# the population of a county doesn't appear to correlate to any specific religious group# the graph below shows the population on the x axis, the percent of population that's religious,# and its facet wrapped to show each religion individually. I chose to do this to compare# each religion and shows that there isn't a relationship between population and proportion# of any specific religionplot1 <- df %>%    ggplot(aes(x = pop_2010,               y = religion_percent,               color = county,               text = paste("County:", county,                            "\nPop:", pop_2010,                            "\nReligion %:", percent(religion_percent, 2)))) +    geom_point() +    facet_wrap(~religion) +    labs(x = "Population 2010", y = "% Population")ggsave("./plot1.png", plot1, bg = 'white')# I wanted to see information about each point so I added some hover boxes on my graphsinteractive_plot1 <- ggplotly(plot1, tooltip = "text")interactive_plot1saveWidget(interactive_plot1, "interactive_plot1.html")# Does proportion of any specific religion in a given county correlate with the proportion of non-religious people?# It looks like the the proportion of of LDS is correlated to the proportion of non-religious# typically as the proportion LDS goes up the proportion non-religious goes down,# showing an inverse relationship.plot2 <- df %>%    ggplot(aes(x = religion,               y = religion_percent,               color = religion,               fill = religion,               text = paste("Religion:", religion,                            "\nPercent:", percent(religion_percent, 2)))) +    geom_col() +    facet_wrap(~county) +    theme_minimal() +    theme(axis.text.x = element_text(angle = 90)) +    labs(x = "Religion", y = "Percent")ggsave("./plot2.png", plot2, bg = 'white')interactive_plot2 <- ggplotly(plot2, tooltip = "text")interactive_plot2saveWidget(interactive_plot2, "interactive_plot2.html")# Looking at the data i wanted to ask if there was any relationship between# the number of different religion in a county and its population# get counts of religions per county where the proportion is > 0df <- df %>%    # creates a column of true false values    mutate(religion_in_county = case_when(religion_percent == 0 ~ FALSE,                                          TRUE ~ TRUE)) %>%    group_by(county) %>%    # counts number of True values per county    mutate(counts_religions = sum(religion_in_county, na.rm=TRUE))plot3 <- df %>%    ggplot(aes(x = pop_2010, y = counts_religions)) +    geom_point(size = 3, aes(color = county)) +    geom_smooth() +    labs(x = "Population", y = "Number Different Religions") +    theme_minimal()ggsave("./plot3.png", plot3, bg = 'white')interactive_plot3 <- ggplotly(plot3)interactive_plot3saveWidget(interactive_plot3, "interactive_plot3.html")
